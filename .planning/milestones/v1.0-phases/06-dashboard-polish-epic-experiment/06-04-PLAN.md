    The chip uses Alpine inline `x-data` with reactive state:
    ```html
    <div
      x-data="{
        state: 'live',
        label: 'LIVE',
        color: 'success',
        update() {
          const isOnline = navigator.onLine;
          const updatedAt = $store.bulkdata?.updatedAt || null;
          if (!isOnline) { this.state = 'offline'; this.label = 'OFFLINE'; this.color = 'secondary'; return; }
          if (updatedAt && (Date.now() - new Date(updatedAt).getTime()) > 86400000) {
            const hours = Math.floor((Date.now() - new Date(updatedAt).getTime()) / 3600000);
            this.state = 'stale'; this.label = 'STALE ' + hours + 'H'; this.color = 'warning'; return;
          }
          this.state = 'live'; this.label = 'LIVE'; this.color = 'success';
        }
      }"
      x-init="update(); window.addEventListener('online', () => { update(); $dispatch('cf:reconnect'); }); window.addEventListener('offline', () => update()); setInterval(() => update(), 60000);"
      class="flex items-center gap-xs px-sm py-xs"
      :title="'Last updated: ' + ($store.bulkdata?.updatedAt ? new Date($store.bulkdata.updatedAt).toLocaleString() : 'Unknown')"
    >
      <span
        class="w-2 h-2 rounded-full"
        :class="{
          'bg-success': color === 'success',
          'bg-warning': color === 'warning',
          'bg-secondary': color === 'secondary'
        }"
      ></span>
      <span
        class="font-mono text-[11px] font-bold uppercase tracking-[0.15em]"
        :class="{
          'text-success': color === 'success',
          'text-warning': color === 'warning',
          'text-secondary': color === 'secondary'
        }"
        x-text="label"
      ></span>
    </div>
    ```

    ```javascript
    window.addEventListener('cf:reconnect', async () => {
      // Check if prices are stale (>24h)
      const { Alpine } = await import('alpinejs');
      const updatedAt = Alpine.store('bulkdata')?.updatedAt;
      if (!updatedAt) return;
      const hoursSince = (Date.now() - new Date(updatedAt).getTime()) / 3600000;
      if (hoursSince > 24) {
        console.log('[Connectivity] Prices stale, triggering background refresh');
        // Import and trigger bulk data pipeline (dynamic import to avoid circular deps)

    });
    ```

      ```javascript
      async removeCard(deckCardId) {
        if (!this.activeDeck) return;
        const deckCard = await db.deck_cards.get(deckCardId);
        if (!deckCard) return;
        const card = await db.cards.get(deckCard.scryfall_id);
        const cardName = card?.name || 'card';
        const deckName = this.activeDeck.name;

        // Remove from UI immediately
        this.activeCards = this.activeCards.filter(c => c.id !== deckCardId);

        Alpine.store('undo').push(
          'deck_card_remove',
          deckCard,
          `Removed ${cardName} from ${deckName}.`,
          async () => {
            await db.deck_cards.delete(deckCardId);
            await db.decks.update(this.activeDeck.id, { updated_at: new Date().toISOString() });
            logActivity('deck_edited', `Removed ${cardName} from "${deckName}"`, deckCard.scryfall_id);
          },
          async () => {
            // Restore: re-add to DB and reload
            await db.deck_cards.add(deckCard);

          }
        );
      },
      ```
    - In `deleteDeck()`, wrap with undo:
      ```javascript
      async deleteDeck(deckId) {
        const deck = await db.decks.get(deckId);
        if (!deck) return;
        const deckCards = await db.deck_cards.where('deck_id').equals(deckId).toArray();
        const deckName = deck.name;

        // Remove from UI immediately
        this.decks = this.decks.filter(d => d.id !== deckId);
        if (this.activeDeck?.id === deckId) {
          this.activeDeck = null;
          this.activeCards = [];
        }

        Alpine.store('undo').push(
          'deck_delete',
          { deck, deckCards },
          `Deleted deck "${deckName}".`,
          async () => {
            await db.transaction('rw', [db.decks, db.deck_cards], async () => {
              await db.deck_cards.where('deck_id').equals(deckId).delete();
              await db.decks.delete(deckId);
            });
            logActivity('deck_edited', `Deleted deck "${deckName}"`);
          },
          async () => {
            // Restore: re-add deck and cards
            await db.decks.add(deck);
            if (deckCards.length > 0) await db.deck_cards.bulkAdd(deckCards);
            await this.loadDecks();
          }
        );
      },
      ```
