    **Step 1 — Create `scripts/assert-bundle-budget.js`:**

    ```javascript
    #!/usr/bin/env node
    /**
     * Phase 13 Plan 5 — bundle size-gate.
     *
     * Inspects dist/ after `npm run build` and fails if any chunk exceeds its budget.
     * Budgets are gzip estimates — we approximate as raw_size * 0.30 for JS, * 0.25 for CSS
     * (these are empirical ratios for minified production output).
     *
     * Wired via `npm run build:check` — run this manually or in CI after any change
     * affecting bundle shape.
     */
    import { readdirSync, statSync, readFileSync } from 'node:fs';
    import { gzipSync } from 'node:zlib';
    import { join } from 'node:path';

    const DIST_ASSETS = 'dist/assets';

    // KB gzipped
    const BUDGETS = {
      main: 300,
      'mana-font': 120,
      keyrune: 50,
      screen: 40,
      vendor: 100,
      default: 500, // Fail loudly if something unexpectedly huge slips through
    };

    function classifyChunk(name) {
      if (/mana-font/.test(name)) return 'mana-font';
      if (/keyrune/.test(name)) return 'keyrune';
      if (/^screen-/.test(name) || /src-screens/.test(name)) return 'screen';
      if (/index-/.test(name) || /^main-/.test(name)) return 'main';
      if (/vendor|chunk/.test(name)) return 'vendor';
      return 'default';
    }

    function toGzipKB(filepath) {
      const raw = readFileSync(filepath);
      const gz = gzipSync(raw);
      return gz.length / 1024;
    }

    function main() {
      try {
        readdirSync(DIST_ASSETS);
      } catch {
        console.error(`[bundle-budget] ${DIST_ASSETS} not found — run 'npm run build' first`);
        process.exit(1);
      }

      const files = readdirSync(DIST_ASSETS).filter(f => /\.(js|css)$/.test(f));
      const violations = [];
      const report = [];

      for (const file of files) {
        const path = join(DIST_ASSETS, file);
        const category = classifyChunk(file);
        const gzipKB = toGzipKB(path);
        const budget = BUDGETS[category] ?? BUDGETS.default;
        const status = gzipKB > budget ? 'FAIL' : 'ok';
        report.push({ file, category, gzipKB: gzipKB.toFixed(1), budget, status });
        if (status === 'FAIL') {
          violations.push({ file, category, gzipKB, budget });
        }
      }

      console.table(report);

      if (violations.length > 0) {
        console.error('\n[bundle-budget] VIOLATIONS:');
        for (const v of violations) {
          console.error(`  - ${v.file} (${v.category}) = ${v.gzipKB.toFixed(1)}KB gzipped exceeds budget ${v.budget}KB`);
        }
        process.exit(1);
      }

      console.log('\n[bundle-budget] All chunks within budget ✓');
      process.exit(0);
    }

    main();
    ```

    Make executable. Note: script uses ESM (`import`) — ensure `package.json` has `"type": "module"` OR rename to `.mjs` extension.
