    const HAS_TEST_USER = !!(process.env.VITE_TEST_USER_EMAIL && process.env.VITE_TEST_USER_PASSWORD);

    describeIf('sync push + propagation (HAS_ENV + HAS_TEST_USER)', () => {
      // Skip if credentials not configured
      if (!HAS_TEST_USER) { test.skip('test user credentials not configured', () => {}); return; }

      let clientA, clientB, userId;

      beforeAll(async () => {
        const { createClient } = await import('@supabase/supabase-js');
        clientA = createClient(URL, KEY, { auth: { persistSession: false } });
        clientB = createClient(URL, KEY, { auth: { persistSession: false } });

          email: process.env.VITE_TEST_USER_EMAIL,
          password: process.env.VITE_TEST_USER_PASSWORD
        });

        userId = authA.user.id;

        // Sign in same user on clientB (or a second household member if Sharon's creds are available)
        await clientB.auth.signInWithPassword({
          email: process.env.VITE_TEST_USER_EMAIL,
          password: process.env.VITE_TEST_USER_PASSWORD
        });
      }, 30000);

      afterAll(async () => {
        if (clientA) await clientA.auth.signOut();
        if (clientB) await clientB.auth.signOut();
      });

      test('push upsert — authenticated user row persists with correct user_id', async () => {
        const id = `test-${RUN}-${Date.now()}`;
        const { error: upErr } = await clientA.schema('counterflux').from('collection').upsert({
          id, user_id: userId, scryfall_id: 'test-scryfall-id', category: 'test', foil: false, quantity: 1
        });
        expect(upErr).toBeNull();

        const { data, error: selErr } = await clientA.schema('counterflux').from('collection').select('*').eq('id', id);
        expect(selErr).toBeNull();
        expect(data?.length).toBe(1);
        expect(data[0].user_id).toBe(userId);

        // Cleanup
        await clientA.schema('counterflux').from('collection').delete().eq('id', id);
      });

      test('Device A → Device B propagation via Realtime (household-scoped)', async () => {
        const id = `test-${RUN}-${Date.now() + 1}`;
        let captured = null;

        // Subscribe client B
        const channel = clientB
          .channel(`test-household-${RUN}`)
          .on('postgres_changes',
            { event: '*', schema: 'counterflux', table: 'collection' },
            (payload) => { if (payload.new?.id === id) captured = payload; }
          );

        await new Promise((resolve, reject) => {
          channel.subscribe((status, err) => {
            if (status === 'SUBSCRIBED') resolve();
            if (status === 'CHANNEL_ERROR') reject(err);
          });
        });

        // Client A writes
        await clientA.schema('counterflux').from('collection').upsert({
          id, user_id: userId, scryfall_id: 'test-realtime-probe', category: 'test', foil: false, quantity: 1
        });

        // Wait up to 5s for propagation
        for (let i = 0; i < 50 && !captured; i++) {
          await new Promise(r => setTimeout(r, 100));
        }

        expect(captured).not.toBeNull();
        expect(captured.new.id).toBe(id);

        // Cleanup
        await channel.unsubscribe();
        await clientA.schema('counterflux').from('collection').delete().eq('id', id);
      }, 15000);
    });

        const { error } = await outsider.schema('counterflux').from('sync_conflicts').select('*').limit(1);
        const acceptableErrors = ['PGRST106', '42501', 'PGRST205', 'PGRST204'];
        expect(error?.code && acceptableErrors.includes(error.code)).toBe(true);
      });
    });
