```js
function colourDistFromPie(colourPie) {
  const total = Object.values(colourPie).reduce((a, b) => a + b, 0);
```

```js
/**
 * Three-tier RAG thresholds per category. Source: 09-RESEARCH.md §"DECK-03
 * Per-Category Dynamic RAG Thresholds" — derived from EDHREC + Draftsim + the
 * Burgess formula (lands) + Frank Karsten (ramp/draw).
 *
 * Each entry: { green: N, amber: N } where:
 *   - count >= green        → severity 'green' (NOT rendered as a badge)
 *   - amber <= count < green → severity 'amber' (rendered as `[AMBER] +N`)
 *   - count < amber          → severity 'red'   (rendered as `[RED] +N`)
 *   - suggestedAdd = green - count (always the gap to GREEN, not AMBER, per RESEARCH)
 */
export const RAG_THRESHOLDS = {
  Ramp:        { green: 10, amber: 6 },
  'Card Draw': { green: 10, amber: 6 },
  Draw:        { green: 10, amber: 6 }, // alias — tag breakdown variation
  Removal:     { green: 8,  amber: 6 },
  'Board Wipe':{ green: 3,  amber: 2 },
  Lands:       { green: 36, amber: 33 },
};

/**
 * Per RESEARCH §3 — creature archetype-aware threshold lookup.
 * @param {string[]} deckTags
 * @returns {{green: number, amber: number}}
 */
export function getCreatureThresholds(deckTags = []) {
  const tags = (deckTags || []).map(t => String(t).toLowerCase());
  if (tags.some(t => ['tribal', 'creatures', 'aggro'].includes(t))) {
    return { green: 30, amber: 20 };
  }
  if (tags.some(t => ['spellslinger', 'control', 'combo'].includes(t))) {
    return { green: 12, amber: 8 };
  }
  return { green: 20, amber: 12 };
}

/**
 * Three-tier RAG gap detection.
 * @param {Object} analytics - computeDeckAnalytics output
 * @param {Object} ragThresholds - RAG_THRESHOLDS or override map
 * @param {string[]} deckTags - for creature archetype branching
 * @param {number} deckSize
 * @returns {Array<{category: string, count: number, threshold: number, severity: 'red'|'amber'|'green', suggestedAdd: number}>}
 */
export function detectGapsRAG(analytics, ragThresholds = RAG_THRESHOLDS, deckTags = [], deckSize = 100) {
  const scale = deckSize / 100;
  const gaps = [];

  // Standard categories
  for (const [category, { green, amber }] of Object.entries(ragThresholds)) {
    const scaledGreen = Math.round(green * scale);
    const scaledAmber = Math.round(amber * scale);
    const count = (category === 'Lands')
      ? (analytics.typeBreakdown?.Land || 0)
      : (analytics.tagBreakdown?.[category] || 0);

    let severity;
    if (count >= scaledGreen) severity = 'green';
    else if (count >= scaledAmber) severity = 'amber';
    else severity = 'red';

    gaps.push({
      category,
      count,
      threshold: scaledGreen,
      severity,
      suggestedAdd: Math.max(0, scaledGreen - count),
    });
  }

  // Creature category — archetype-aware
  const creatureThresholds = getCreatureThresholds(deckTags);
  const creatureCount = analytics.typeBreakdown?.Creature || 0;
  const cGreen = Math.round(creatureThresholds.green * scale);
  const cAmber = Math.round(creatureThresholds.amber * scale);
  let cSeverity;
  if (creatureCount >= cGreen) cSeverity = 'green';
  else if (creatureCount >= cAmber) cSeverity = 'amber';
  else cSeverity = 'red';
  gaps.push({
    category: 'Creatures',
    count: creatureCount,
    threshold: cGreen,
    severity: cSeverity,
    suggestedAdd: Math.max(0, cGreen - creatureCount),
  });

  // Sort: red first, then amber, then green
  const order = { red: 0, amber: 1, green: 2 };
  gaps.sort((a, b) => order[a.severity] - order[b.severity]);

  return gaps;
}
```

**Step 2 — Add the `resolveCommanderCard` helper (top of the file or in a `utils` module — keep local for now to minimise file churn):**

```js
/**
 * Resolve the commander card for the active deck.
 * Per D-07 + P-9 fallback: prefer activeDeck.commander_id; otherwise derive
 * from the first Legendary Creature/Planeswalker whose color_identity is a
 * subset of the deck's union. Returns null if no candidate found.
 *
 * @param {Object} store - Alpine deck store (has activeDeck + activeCards)
 * @returns {Object|null} Card object from activeCards, or null
 */
function resolveCommanderCard(store) {
  if (!store?.activeDeck || !store?.activeCards) return null;
  const cards = store.activeCards;
  if (cards.length === 0) return null;

  // Primary: activeDeck.commander_id (Phase 7 v8 row field)
  if (store.activeDeck.commander_id) {
    const found = cards.find(c => c.id === store.activeDeck.commander_id || c.scryfall_id === store.activeDeck.commander_id);
    if (found) return found;
  }

  // Fallback: first Legendary Creature/Planeswalker matching deck's colour identity
  const deckColours = new Set();
  for (const c of cards) {
    for (const colour of (c.color_identity || [])) deckColours.add(colour);
  }
  const legendary = cards.find(c => {
    const tl = (c.type_line || '').toLowerCase();
    if (!tl.includes('legendary')) return false;
    if (!tl.includes('creature') && !tl.includes('planeswalker')) return false;
    const cardColours = c.color_identity || [];
    return cardColours.every(col => deckColours.has(col));
  });
  if (legendary) {
    console.warn('[deck-centre-panel] commander_id missing; derived fallback:', legendary.name);
    return legendary;
  }
  return null;
}
```
