    **Step 1 — Create `src/components/topbar-bulkdata-pill.js`:**

    Export an Alpine component factory that returns the data object + any computed methods. The TEMPLATE itself lives in `index.html` (single-source, Alpine reads x-data from the factory). Pattern example:

    ```javascript
    /**
     * D-06 topbar bulk-data progress pill (Phase 13 Plan 3).
     *
     * Renders adjacent to the sync-status chip + notification bell while
     * `$store.bulkdata.status !== 'ready'`. Auto-dismisses when status flips to 'ready'.
     *
     * Visual vocabulary inherited from the sync-status chip — NO new design tokens,
     * colours, or keyframes (D-06; Research §Pattern 3).
     *
     * States (Research §Example 6 template):
     * - `checking` / `idle` → CHECKING ARCHIVE… (muted)
     * - `downloading`       → ARCHIVE — N% (primary tint, spinner glyph)
     * - `parsing`           → INDEXING — N CARDS (primary tint, live dot)
     * - `error`             → ARCHIVE ERROR (button, warning tint, clickable → retry)
     */
    export function topbarBulkdataPill() {
      return {
        // Alpine reads these as reactive getters via $store.bulkdata — no local state needed.
        // Component factory returns an empty object so Alpine.data() registration succeeds,
        // but all template bindings flip through $store.bulkdata.{status|progress|...}.
        //
        // Keep as a function so future state can be added without touching the template.
        get status() {
          return window.Alpine?.store('bulkdata')?.status ?? 'idle';
        },
        get progressLabel() {
          const store = window.Alpine?.store('bulkdata');
          if (!store) return '';
          if (store.status === 'downloading') return `${store.progress ?? 0}%`;
          if (store.status === 'parsing') return `${(store.parsed ?? 0).toLocaleString()}`;
          return '';
        },
        retry() {
          const store = window.Alpine?.store('bulkdata');
          if (store && typeof store.retry === 'function') store.retry();
        },
      };
    }
    ```
