       ```javascript
       import Sortable from 'sortablejs';

       // After rendering search results list container:
       const searchSortable = new Sortable(resultsListEl, {
         group: {
           name: 'deck-cards',
           pull: 'clone',   // Clone from search, don't remove original
           put: false        // Don't allow drops into search
         },
         sort: false,         // Search results stay in order
         ghostClass: 'drag-ghost',
         onEnd(evt) {
           // Remove the cloned DOM element (store handles rendering)
           evt.item.remove();
           const scryfallId = evt.item.dataset.scryfallId;
           if (scryfallId && evt.to !== evt.from) {
             Alpine.store('deck').addCard(scryfallId);
           }
         }
       });
       ```
       Destroy and re-init the SortableJS instance when search results change (new query).
