State machine (RESEARCH §Pattern 7):
| Event                                          | From      | To                |
|------------------------------------------------|-----------|-------------------|
| enqueue() called                               | synced    | syncing           |
| enqueue() called                               | offline   | offline (stays)   |
| flushQueue completes clean                     | syncing   | synced            |
| flushQueue hits permanent error                | syncing   | error             |
| flushQueue hits transient error, attempts < 3  | syncing   | syncing           |
| navigator.online → false                       | *         | offline           |
| navigator.online → true + authed               | offline   | syncing           |
| auth.status → 'anonymous'                      | *         | store resets      |
| User clicks RETRY in sync-errors-modal         | error     | syncing           |

    ```js
    // src/stores/sync.js
    //
    // Phase 11 Plan 2 — Alpine sync store + 4-state state machine (SYNC-07, D-08).
    //
    // State machine (RESEARCH §Pattern 7):
    //   - Initial: 'synced' if online + no pending, 'offline' if navigator.onLine false
    //   - Events: enqueue / flush-ok / flush-err-permanent / online / offline / auth-anonymous / retry
    //
    // Scope in this plan:
    //   - Store shape + state machine wiring
    //   - navigator.online/offline listeners
    //   - STUB methods for flush/retry/discard — Plan 11-04 (engine) and Plan 11-03 (sync-errors modal open-handler) land the real impl
    //
    // Design notes:
    //   - Module-level subscription flag (_onlineListenerInstalled) prevents double-bind across init calls
    //   - Pre-auth boot: chip should not render anyway (auth-wall hides topbar per Phase 10 D-40);
    //     store status remains 'synced' as a harmless default until auth flips authed
    //   - pending_count is computed from db.sync_queue in Plan 11-04; here we expose the reactive field only

    import Alpine from 'alpinejs';

    let _onlineListenerInstalled = false;

    // Opens sync-errors modal — stubbed here for chip error-state click binding.
    // Plan 11-03 will replace this with the real import from src/components/sync-errors-modal.js.
    function openSyncErrorsModalStub() {
      console.warn('[sync] sync-errors modal requested but Plan 11-03 not yet shipped');
    }

    // Expose globally so index.html chip @click can reach it without tight coupling.
    if (typeof window !== 'undefined') {
      window.openSyncErrorsModal = window.openSyncErrorsModal || openSyncErrorsModalStub;
    }

    const VALID_TRANSITIONS = {
      synced:  ['syncing', 'offline', 'error'],
      syncing: ['synced', 'offline', 'error'],
      offline: ['syncing'],
      error:   ['syncing', 'synced']   // retry/discard-all resolves
    };

    export function initSyncStore() {
      Alpine.store('sync', {
        status: (typeof navigator !== 'undefined' && navigator.onLine === false) ? 'offline' : 'synced',
        pending_count: 0,
        last_error: null,
        last_synced_at: null,
        bulkPullProgress: null,

        async init() {
          // Install online/offline listeners exactly once
          if (_onlineListenerInstalled || typeof window === 'undefined') return;
          _onlineListenerInstalled = true;

          window.addEventListener('online', () => {
            const auth = Alpine.store('auth');
            if (auth?.status === 'authed') {
              this._transition('syncing');
              // Plan 11-04 wires: flushQueue triggers here
            } else {
              this._transition('synced');
            }
          });

          window.addEventListener('offline', () => {
            this._transition('offline');
          });
        },

        _transition(next) {
          if (this.status === next) return;
          const allowed = VALID_TRANSITIONS[this.status] ?? [];
          if (!allowed.includes(next)) {
            console.warn(`[sync] invalid transition ${this.status} → ${next}; ignoring`);
            return;
          }
          this.status = next;
          if (next === 'synced') this.last_synced_at = Date.now();
        },

        async flush() {
          // STUB — Plan 11-04 wires src/services/sync-engine.js flushQueue()
          console.info('[sync] flush() stub — Plan 11-04 pending');
        },

        async retry(queueEntryId) {
          // STUB — Plan 11-04 wires sync-engine retry path
          console.info('[sync] retry() stub —', queueEntryId);
          this._transition('syncing');
        },

        async discard(queueEntryId) {
          // STUB — Plan 11-04 wires sync-engine discard path
          console.info('[sync] discard() stub —', queueEntryId);
        },

        // Status tooltip helper used by the chip template
        getTooltip() {
          switch (this.status) {
            case 'synced': {
              if (!this.last_synced_at) return 'Last synced just now.';
              const t = new Date(this.last_synced_at).toTimeString().slice(0, 8);
              return `Last synced ${t}.`;
            }
            case 'syncing': return `${this.pending_count} pending changes.`;
            case 'offline': return `No connection. Changes saved locally; will sync when you're back online.`;
            case 'error':   return `Sync failed. Click to review.`;
            default:        return '';
          }
        }
      });

      // Kick off listeners on next microtask so Alpine.start() has completed
      Promise.resolve().then(() => Alpine.store('sync').init());
    }

    // Test helper — exported for tests/sync-store.test.js to reset listener flag
    export function __resetSyncStoreForTests() {
      _onlineListenerInstalled = false;
    }
    ```
